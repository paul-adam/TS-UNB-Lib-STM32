174  uint16_t encode(uint8_t* const mpduPayload, const uint8_t* const macPayload, const uint16_t len,
175  const bool MPF_present = false, const uint8_t MPF_value = 0) {
176  // Create AES on the stack
177  TsUnb::Aes128 Aes;
178  Aes_p = &Aes;
179  Aes_p->init(networkKey);
180 
181  // Set MPF field in header
182  macHeader.bit.mpfflag = MPF_present;
183 
184  // CMAC initlization vector
185  // EUI64
186  uint8_t iv[BLOCK_SIZE_AES];
187  for(uint8_t i = 0; i < 8; ++i)
188  iv[i] = eui64[i];
189  iv[8] = 0x00u;
190  iv[9] = DATA_DIRECTION;
191  iv[10] = extPkgCnt >> 24;
192  iv[11] = extPkgCnt >> 16;
193  iv[12] = extPkgCnt >> 8;
194  iv[13] = extPkgCnt;
195  iv[14] = 0xFFu;
196  iv[15] = 0xFFu;
197 
198 
199  // Actual packet
200  uint16_t idx = 0;
201  mpduPayload[idx++] = macHeader.reg;
202  if(macHeader.bit.addressingflag) // Long address
203  {
204  for(uint8_t i = 0; i < 8; ++i)
205  mpduPayload[idx++] = eui64[i];
206  }
207  else
208  {
209  for(uint8_t i = 0; i < 2; ++i)
210  mpduPayload[idx++] = shortAddr[i];
211  }
212  mpduPayload[idx++] = extPkgCnt >> 16;
213  mpduPayload[idx++] = extPkgCnt >> 8;
214  mpduPayload[idx++] = extPkgCnt;
215  uint16_t beginEncrypted = idx;
216 
217  // We never use a MPF header
218  if(macHeader.bit.mpfflag)
219  mpduPayload[idx++] = MPF_value;
220 
221  for(uint8_t i = 0; i < len; ++i)
222  mpduPayload[idx++] = macPayload[i];
223 
224 
225  // Encryption with IV as input
226  for(uint8_t block = 0; beginEncrypted < idx; ++block)
227  {
228  iv[14] = 0; // Block counter will never exceed one byte
229  iv[15] = block;
230  uint8_t ivEnc[BLOCK_SIZE_AES];
231  Aes_p->chipher(iv, ivEnc);
232 
233  for(uint8_t i = 0; (i < BLOCK_SIZE_AES) && (beginEncrypted < idx); ++i)
234  mpduPayload[beginEncrypted++] ^= ivEnc[i];
235  }
236 
237  iv[14] = 0xFFu;
238  iv[15] = 0xFFu;
239 
240  cmacGenerateTag(iv, mpduPayload, idx, iv);
241 
242  for(uint8_t i = 0; i < 4; ++i)
243  mpduPayload[idx++] = iv[i];
244  extPkgCnt++;
245  return idx;
246 
247  }

422  void cmacGenerateSubkey(uint8_t *subkey1, uint8_t *subkey2)
423  {
424  for(uint8_t i = 0; i < BLOCK_SIZE_AES; ++i)
425  subkey1[i] = 0;
426  Aes_p->chipher(subkey1, subkey1);
427 
428  if(leftShift16byte(subkey1, subkey1))
429  subkey1[BLOCK_SIZE_AES - 1] ^= CMAC_RB;
430  if(leftShift16byte(subkey1, subkey2))
431  subkey2[BLOCK_SIZE_AES - 1] ^= CMAC_RB;
432  }

434  uint8_t leftShift16byte(const uint8_t *dataIn, uint8_t *dataOut) const {
435  uint8_t prevMsb = 0;
436  for(int8_t i = 15; i >= 0; --i)
437  {
438  uint8_t msb = dataIn[i] >> 7;
439  dataOut[i] = (dataIn[i] << 1) | prevMsb;
440  prevMsb = msb;
441  }
442  return prevMsb;
443  }

446  void cmacGenerateTag(const uint8_t* const input, const uint16_t inputLen, uint8_t* const output) {
447  uint8_t subkey1[BLOCK_SIZE_AES], subkey2[BLOCK_SIZE_AES];
448 
449  cmacGenerateSubkey(subkey1, subkey2);
450 
451  uint8_t blocks = (uint16_t)(inputLen + BLOCK_SIZE_AES - 1) >> 4;
452  uint8_t flag;
453 
454  if(blocks == 0) {
455  blocks = 1;
456  flag = 0;
457  }
458  else {
459  if(inputLen & 0x000Fu)
460  flag = 0;
461  else
462  flag = 1;
463  }
464 
465  for(uint8_t i = 0; i < BLOCK_SIZE_AES; ++i)
466  output[i] = 0;
467 
468  for(uint16_t msg = 0; msg < (uint16_t)(blocks - 1); ++msg)
469  {
470  xor16byte(&input[msg << 4], output);
471  Aes_p->chipher(output,output);
472  }
473 
474  if(flag) {
475  xor16byte(subkey1, output);
476  xor16byte(&input[(blocks-1) << 4], output);
477  }
478  else {
479  xor16byte(subkey2, output);
480  uint8_t extraBytes = inputLen & 0x000Fu;
481  for(uint8_t i = 0; i < BLOCK_SIZE_AES; ++i) {
482  if(i < extraBytes)
483  output[i] ^= input[((blocks-1) << 4) + i];
484  else if(i == extraBytes)
485  output[i] ^= 0x80u;
486  else
487  output[i] ^= 0x00u;
488  }
489  }
490  Aes_p->chipher(output,output);
491  }
