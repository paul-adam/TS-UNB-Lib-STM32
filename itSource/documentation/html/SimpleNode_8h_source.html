159  int16_t send(const uint8_t* const payload, const uint16_t payloadLength,
160  const uint8_t MPF_value = 0, const bool priority= false) {
161 
163  const bool MPF_present = MPF_value != 0;
164 
165  uint16_t MPDU_length = Mac.MPDU_Length(payloadLength, MPF_present);
166 
167  if (MPDU_length == 0)
168  return -1;
169 
170  uint8_t MPDU[MPDU_length];
171  Mac.encode(MPDU, payload, payloadLength, MPF_present, MPF_value);
172 
174  PHY Phy;
175 
176  uint16_t numRadioBursts = Phy.numRadioBursts(MPDU_length);
177  if (SYNC_BURST == true)
178  numRadioBursts++;
179 
181  typename PHY::RadioBurst_t Bursts[numRadioBursts];
182 
183  // Transmit frequency
184  uint32_t freqReg;
185 
186  // We have to do a seperate handling if the Sync Burts is used
187  if (SYNC_BURST == false) {
188  // Normal mode without sync burst
189  if (priority)
190  freqReg = Phy.encode(Bursts, MPDU, MPDU_length, 6);
191  else
192  freqReg = Phy.encode(Bursts, MPDU, MPDU_length, Mac.getTsmaPattern());
193 
194  if (freqReg > 0)
195  return Tx.transmit(Bursts, numRadioBursts, freqReg);
196  else
197  return -1;
198  }
199  else {
200  // This is special handling in caes of a sync burst
201  numRadioBursts += 1;
202 
203  // The first data Burst is Burst[1] as Burst[0] is the Sync Burst
204  if (priority) {
205  freqReg = Phy.encode(&Bursts[1], MPDU, MPDU_length, 6);
206  Phy.encodeSyncBurst(&Bursts[0], 6, Mac.shortAddr[1]);
207  }
208  else {
209  const uint8_t tsmaPattern = Mac.getTsmaPattern();
210  freqReg = Phy.encode(&Bursts[1], MPDU, MPDU_length, tsmaPattern);
211  Phy.encodeSyncBurst(&Bursts[0], tsmaPattern, Mac.shortAddr[1]);
212  }
213 
214  }
215  if (freqReg > 0)
216  return Tx.transmit(Bursts, numRadioBursts, freqReg);
217  else
218  return -1;
219 
220  }
221 
223  TX Tx;
224 
226  MAC Mac;
