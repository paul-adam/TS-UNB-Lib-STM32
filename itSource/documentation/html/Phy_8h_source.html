244  uint32_t encode(RadioBurst_T* const RadioBursts, const uint8_t* const MPDU,
245  const uint16_t MPDU_Length, const uint8_t TSMAPattern = 0) {
246 
247  if (MPDU_Length > TSUNBPHY_MAX_PSDU_LENGTH)
248  return 0;
249 
250  const uint16_t numBursts = numRadioBursts(MPDU_Length);
251 
252 
253  /*
254  * Copy data to local buffer and set fields
255  */
256  uint8_t PhyPayload[numBursts];
257  for (uint16_t i = 0; i < MPDU_Length; ++i) {
258  PhyPayload[TSUNBPHY_PAYLOAD_DATA_POS + i] = MPDU[i];
259  }
260  PhyPayload[TSUNBPHY_PAYLOAD_PSI_POS] = (uint8_t) MPDU_Length;
261 
262 
263  // The MMODE is copied at the end of the payload data for CRC calculation.
264  // It is copied to the correct position if stuffing is required later.
265  PhyPayload[TSUNBPHY_PAYLOAD_DATA_POS + MPDU_Length] = MMODE << 6;
266 
267  // Calculate the payload CRC
268  PhyPayload[TSUNBPHY_PAYLOAD_CRC_POS] =
269  calcCRC8(&PhyPayload[TSUNBPHY_PAYLOAD_DATA_POS], MPDU_Length * 8 + 2);
270 
271 
272  // Stuff in case of short PSDU and bring the MMODE to the right position
273  if (MPDU_Length < TSUNBPHY_MIN_PSDU_LENGTH) {
274  // We have to stuff the data
275  for (uint16_t i = MPDU_Length;i < TSUNBPHY_MIN_PSDU_LENGTH; ++i) {
276  PhyPayload[TSUNBPHY_PAYLOAD_DATA_POS + i] = 0;
277  }
278 
279  // Finally copy the MMODE to the right position at the end of the stuffing data
280  PhyPayload[TSUNBPHY_PAYLOAD_DATA_POS + TSUNBPHY_MIN_PSDU_LENGTH] = MMODE << 6;
281  }
282 
283  // Calculate the header CRC
284  PhyPayload[TSUNBPHY_HEADER_CRC_POS] = calcCRC8(&PhyPayload[TSUNBPHY_PAYLOAD_CRC_POS], 16);
285 
286 
288  const uint8_t payloadCrc = PhyPayload[TSUNBPHY_PAYLOAD_CRC_POS];
289 
291  uint16_t lfsrSeed =
292  0x8000u | (uint16_t) PhyPayload[TSUNBPHY_HEADER_CRC_POS] << 8 | payloadCrc;
293 
294 
295  /*
296  * Whiten the data
297  */
298  whitenData(PhyPayload, numBursts);
299 
300  /*
301  * Do the convolutional encoding.
302  * The coding is quite complex due to the bit shift in the interleaver.
303  * To minimize memory consumption the interleaving and the convolutional
304  * encoding are done in a single step.
305  */
307  const uint16_t payloadBits = numBursts * 8;
308 
310  uint8_t convReg = 0;
311 
312  // The code termination is achieved by means of the zero bits in the MMODE field.
313  // Therefore we have to restore our tail bits that we lost during the whitening.
314  PhyPayload[numBursts - 1] &= 0xC0;
315 
316 
317 
318  // To avoid an additional memory for the cylic shift of the interleaver we
319  // use some kind of tail biting covolutional code. For this purpose we
320  // first have to bring the code to the correct register state.
321  for (uint16_t inBitIdx = 0; inBitIdx < 8; ++inBitIdx) {
322  convReg <<= 1;
323 
324  const uint16_t shiftBitIdx = payloadBits + inBitIdx - (TSUNBPHY_NUM_BITS_SHIFT / 3 + 8);
325  if (readBit(shiftBitIdx, PhyPayload))
326  convReg |= 1;
327  }
328 
329  // Here we to the actual convolutional encoding
330  // We normally would have to reset the convolution encoder at the end
331  // of the payload data before starting to encode the initial header data,
332  // which is caused by the cyclic shift. However, the MMODE only takes 2 bits
333  // and we have 6 zeros in this byte. This actually terminates the code and
334  // we start again from zero.
335  for (uint16_t inBitIdx = 0; inBitIdx < payloadBits; ++inBitIdx) {
336 
337  // Consider bit shift due to the interleaver and correct if necessary
338  int16_t shiftBitIdx = (int16_t) inBitIdx - TSUNBPHY_NUM_BITS_SHIFT / 3;
339  if (shiftBitIdx < 0)
340  shiftBitIdx += payloadBits;
341 
342  convReg <<= 1;
343  if (readBit(shiftBitIdx, PhyPayload))
344  convReg |= 1;
345 
346 
347  // Get the one output from each branch of the convolutional encoder
348  uint8_t outBits[3];
349  outBits[0] = convEncode_parity(TSUNBPHY_CONV_POLY_G1 & convReg);
350  outBits[1] = convEncode_parity(TSUNBPHY_CONV_POLY_G2 & convReg);
351  outBits[2] = convEncode_parity(TSUNBPHY_CONV_POLY_G3 & convReg);
352 
353  // And write the bits directly onto their correct positions on the radio bursts
354  for (uint8_t i = 0; i < 3; ++i) {
355  uint16_t outBitIdx = inBitIdx * 3 + i;
356  const uint16_t burstIdx = getRadioBurstIdx(outBitIdx, numBursts);
357  RadioBursts[burstIdx].writeSubPacketBit(outBits[i], burstIdx);
358  }
359  }
360 
361 
362  /*
363  * Add the midamble and do the differntial MSK encoding
364  */
365  for (uint16_t burstIdx = 0; burstIdx < numBursts; ++burstIdx) {
366  RadioBursts[burstIdx].addMidamble(burstIdx);
367  RadioBursts[burstIdx].differentialMSKEncoding();
368  }
369 
370 
371  /*
372  * Add the TSMA pattern
373  */
374  if (TSUNB_UPG == TsUnb_UPG3) {
375  addTsmaPattern(numBursts, 0, lfsrSeed, RadioBursts);
376  }
377  else {
378  addTsmaPattern(numBursts, TSMAPattern % TSUNBPHY_UNB_NUM_P,
379  lfsrSeed, RadioBursts);
380  }
381 
382  /*
383  * Return frequency f_0
384  */
385  return calcFreqReg(payloadCrc);
386  }

411  void encodeSyncBurst(RadioBurst_T* const RadioBurst, const uint8_t TSMAPattern, const uint8_t LSB_ShortAddress) {
412 
413  // Data array for the Sync Burst
414  // Note: The TSMAPattern rages from 0 (p=1) to 7 (p=8)
415  uint8_t syncBurst[5] = {0x33, 0x3d, (uint8_t)(0x30 + (TSMAPattern & 0x07)), LSB_ShortAddress, 0};
416  switch (TSUNB_UPG) {
417  case TsUnb_UPG1:
418  break;
419  case TsUnb_UPG2:
420  syncBurst[4] |= 0x40;
421  break;
422  case TsUnb_UPG3:
423  syncBurst[4] |= 0x80;
424  break;
425  }
426 
428  uint8_t crc2_reg = TSUNBPHY_CRC2_INIT;
429 
430  for (uint16_t bitIdx = 20; bitIdx <= 33; ++bitIdx) {
431  // Get msb
432  uint8_t msb = 0;
433  if (crc2_reg & 0x03)
434  msb = 1;
435 
436  // Check if input bit set, if yes xor MSB
437  if (readBit(bitIdx, syncBurst))
438  msb ^= 1;
439 
440  crc2_reg <<= 1;
441 
442  if (msb)
443  crc2_reg ^= TSUNBPHY_CRC2_POLY;
444  }
445 
446  RadioBurst->writeBitIdx((crc2_reg >> 1) & 0x01, 34);
447  RadioBurst->writeBitIdx((crc2_reg >> 0) & 0x01, 35);
448 
449  for (uint8_t bitIdx = 0; bitIdx < RadioBurst_T::BURST_LENGTH; ++bitIdx) {
450  const uint8_t bit = readBit(bitIdx, syncBurst);
451  RadioBurst->writeBitIdx(bit, bitIdx);
452  }
453 
454  RadioBurst->differentialMSKEncoding();
455  RadioBurst->setCarrierOffset((uint16_t)24 * B_c);
456 
457  switch (TSUNB_UPG) {
458  case TsUnb_UPG1:
459  case TsUnb_UPG2:
460  RadioBurst->set_T_RB(337);
461  break;
462  case TsUnb_UPG3:
463  RadioBurst->set_T_RB(66);
464  break;
465  }
466 
467  return;
468  }

482  uint8_t calcCRC8(const uint8_t* const inputBytes, const uint16_t numInputBits) const {
484  uint8_t crc8_reg = TSUNBPHY_CRC8_INIT;
485 
486  for (uint16_t bitIdx = 0; bitIdx < numInputBits; ++bitIdx) {
487  // Get msb
488  uint8_t msb = 0;
489  if (crc8_reg & 0x80)
490  msb = 1;
491 
492  // Check if input bit set, if yes xor MSB
493  if (readBit(bitIdx, inputBytes))
494  msb ^= 1;
495 
496  crc8_reg <<= 1;
497 
498  if (msb)
499  crc8_reg ^= TSUNBPHY_CRC8_POLY;
500  }
501 
502  return crc8_reg;
503  }

514  void whitenData(uint8_t* const inputBytes, const uint16_t numBytes) const {
515  uint16_t reg = 0x1FF;
516 
517  for (uint16_t byte = 0; byte < numBytes; ++byte) {
518  for (uint8_t bit = 0; bit < 8; ++bit) {
519  reg <<= 1;
520  reg ^= 0x1 & (reg >> 9 ^ reg >> 4);
521  }
522  inputBytes[byte] ^= (uint8_t) reg;
523  }
524  }

562  uint16_t getRadioBurstIdx(const uint16_t bitIdx, const uint16_t numPackets) const {
563  if (bitIdx < TSUNBPHY_NUM_BITS_CORE_ILV) { // Core frame (first 288 bit)
564  return bitIdx % TSUNBPHY_NUM_CORE_BURSTS;
565  }
566  else { // Extension frame
567  uint16_t groupIdx = bitIdx - TSUNBPHY_NUM_BITS_CORE_ILV;
568  const uint16_t groupLen =
569  numPackets - (TSUNBPHY_NUM_CORE_BURSTS >> 1);
570 
571  const uint16_t group = groupIdx / groupLen;
572  groupIdx -= group * groupLen;
573 
574  if (groupIdx < (TSUNBPHY_NUM_CORE_BURSTS >> 1)) // First 12 bit
575  return (groupIdx << 1) + (group & 1);
576  else
577  return groupIdx + (TSUNBPHY_NUM_CORE_BURSTS >> 1);
578  }
579  }

594  void addTsmaPattern(const uint16_t numBursts, const uint8_t TSMAPattern,
595  uint16_t lfsrSeed, RadioBurst_T* const RadioBursts) const {
596 
597  for (uint16_t i = 0; i < TSUNBPHY_NUM_CORE_BURSTS; ++i) {
598  RadioBursts[i].setCarrierOffset((uint16_t)get_C_RB(TSMAPattern, i) * B_c);
599 
600  if (i != TSUNBPHY_NUM_CORE_BURSTS - 1)
601  RadioBursts[i].set_T_RB(get_T_RB(TSMAPattern, i));
602  }
603 
604 
605  for (uint16_t i = TSUNBPHY_NUM_CORE_BURSTS; i < numBursts; ++i) {
606  lfsrSeed = tsmaLfsr(lfsrSeed);
607  RadioBursts[i].setCarrierOffset(((lfsrSeed >> 8) % 25) * B_c);
608 
609  uint16_t extentionFrameTimeSpacing;
610 
611  switch (TSUNB_UPG) {
612  case TsUnb_UPG1:
613  extentionFrameTimeSpacing = TSUNBPHY_TIME_SPACING_UPG1;
614  break;
615 
616  case TsUnb_UPG2:
617  extentionFrameTimeSpacing = TSUNBPHY_TIME_SPACING_UPG2;
618  break;
619 
620  case TsUnb_UPG3:
621  extentionFrameTimeSpacing = TSUNBPHY_TIME_SPACING_UPG3;
622  break;
623  }
624 
625  RadioBursts[i - 1].set_T_RB(extentionFrameTimeSpacing + (lfsrSeed % 128));
626  }
627 
628  RadioBursts[numBursts - 1].set_T_RB(0);
629  }

643  uint32_t calcFreqReg(const uint8_t payloadCrc) const {
644  // Calculate frequency according to 6.4.7.1.1
645  uint32_t f_c;
646 
647  if ((payloadCrc & 0x80u) == 0)
648  f_c = CHAN_A;
649  else
650  f_c = CHAN_B;
651 
652  // The only valid values for nco are 3 and 11
653  const int8_t v_c0 = (int8_t) (payloadCrc & 0x7Fu);
654  const int32_t C_RF = (v_c0 % n_co) - (n_co >> 1);
655  const uint32_t f_0 = f_c - 12 * B_c + C_RF * B_c0;
656 
657  return f_0;
658  }
