328  void differentialMSKEncoding() {
329  uint8_t shiftedData = 0;
330  uint8_t firstBitLastByte = 0;
331  for(uint16_t s = 0; s < BURST_LENGTH_BYTES; ++s) {
332  shiftedData = firstBitLastByte | (data[s] >> 1);
333  firstBitLastByte = data[s] << 7;
334  data[s] ^= shiftedData;
335  }
336  // The first and last two bits of each burst are don't cares
337  // Setting at least one initial bit to 1 is a workaround
338  if (HEAD_BITS > 0)
339  data[0] |= 0x80;
340 
341  }
342 
346  void puncture() {
347  carrierOffset = 0xFFFF;
348  }

368  uint16_t getSubPkgBitIdx(const uint16_t burstIdx, const uint8_t bitIdx) const {
369  if((burstIdx ^ bitIdx) & 1) // (pkt even & bit odd) | (pkt odd & bit even)
370  return 24 + (bitIdx >> 1); // 24 offset
371  else // (pkt odd & bit odd) | (pkt even & bit even)
372  return 11 - (bitIdx >> 1); // 11 offset
373  }
