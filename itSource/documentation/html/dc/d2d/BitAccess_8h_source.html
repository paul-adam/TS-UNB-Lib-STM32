98 static inline uint8_t readBit(const uint_fast16_t bitIdx, const uint8_t* const memory) {
99  const uint_fast16_t byteIdx = bitIdx >> 3; // equivalent to / 8;
100  const uint_fast8_t intraByteIdx = bitIdx & 0x07; // equivalent to % 8
101 
102  // Check if bit is set, always shift to MSB
103  if ((memory[byteIdx] << intraByteIdx) & 0x80)
104  return 1;
105  else
106  return 0;
107 }

116 static inline void writeBit(const uint8_t bit, const uint_fast16_t bitIdx, uint8_t* const memory) {
117  const uint_fast16_t byteIdx = bitIdx >> 3; // equivalent to / 8;
118  const uint_fast8_t intraByteIdx = bitIdx & 0x07; // equivalent to % 8
119 
120  if (bit) {
121  memory[byteIdx] |= (0x80 >> intraByteIdx);
122  }
123  else {
124  memory[byteIdx] &= ~(0x80 >> intraByteIdx);
125  }
126 
127  return;
128 }
