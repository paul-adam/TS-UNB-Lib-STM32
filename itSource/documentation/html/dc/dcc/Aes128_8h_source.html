163  void init (const uint8_t* const key) {
164  for (uint16_t i = 0; i < AES_BYTES; ++i)
165  keyW[0][i] = key[i];
166 
167  uint8_t temp[AES_WORD];
168  uint8_t * lastWord;
169 
170  lastWord = &keyW[0][AES_BYTES - AES_WORD];
171 
172  // Will not work for other AES key lengths
173  for (uint16_t roundIdx = 1; roundIdx <= AES_NR; ++roundIdx) {
174  subRotRconWord (lastWord, temp, roundIdx);
175  lastWord = temp;
176 
177  for (uint8_t wordIdx = 0; wordIdx < AES_WORD; ++wordIdx) {
178  xorWord (&keyW[roundIdx - 1][wordIdx << 2],
179  lastWord, &keyW[roundIdx][wordIdx << 2]);
180 
181  lastWord = &keyW[roundIdx][wordIdx << 2];
182  }
183  }
184  }

194  void chipher (const uint8_t* const in, uint8_t* const out) {
195  for (uint8_t i = 0; i < AES_BYTES; ++i)
196  out[i] = in[i];

284  void mixColumns (uint8_t* const data) {
285  uint8_t input[4];
286  uint8_t twoTimesPoly;
287  uint8_t * column;
288 
289  for (uint8_t col = 0; col < AES_NB; ++col) {
290 
291  column = &data[col << 2];
292  for (uint8_t i = 0; i < AES_WORD; ++i) {
293  input[i] = column[i];
294  column[i] = 0;
295  }
296 
297  for (uint8_t inRow = 0; inRow < AES_WORD; ++inRow) {
298  twoTimesPoly = multiplyByX (input[inRow]);
299 
300  column[inRow] ^= twoTimesPoly;
301  column[(inRow + 3) & 3] ^= twoTimesPoly;
302 
303  for (uint8_t outRow = 1; outRow < AES_WORD; ++outRow)
304  column[(inRow + outRow) & 3] ^= input[inRow];
305  }
306  }
307  }

317  uint8_t multiplyByX (const uint8_t polynomial) const {
318  if (polynomial & 0x80) {
319 
320  // Polynomial has to be reduced
321  uint16_t tmp = ((uint16_t) polynomial) << 1;
322  tmp ^= AES_MOD_POLY;
323 
324  return (uint8_t) tmp;
325  }
326  else
327  return polynomial << 1;
328  }

337  void addRoundKey (uint8_t* const data, const uint8_t roundIdx) const {

350  void subRotRconWord (uint8_t* const in, uint8_t* const out,
351  const uint8_t roundIdx) const {
352 
353  uint8_t rconPoly = 1 << (roundIdx - 1);
354 
355  if (roundIdx > 8)
356  // Will not work for other AES key lengths
357  rconPoly ^= AES_MOD_POLY << (roundIdx - 9);
358 #ifdef __AVR_ARCH__
359  out[0] = (uint8_t)pgm_read_byte (&(AES_sBox[in[1]])) ^ rconPoly;
360  out[1] = (uint8_t)pgm_read_byte (&(AES_sBox[in[2]]));
361  out[2] = (uint8_t)pgm_read_byte (&(AES_sBox[in[3]]));
362  out[3] = (uint8_t)pgm_read_byte (&(AES_sBox[in[0]]));
363 #else
364  out[0] = AES_sBox[in[1]] ^ rconPoly;
365  out[1] = AES_sBox[in[2]];
366  out[2] = AES_sBox[in[3]];
367  out[3] = AES_sBox[in[0]];
368 #endif
369  }
