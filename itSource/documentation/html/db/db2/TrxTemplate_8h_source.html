163  int16_t transmit(const RadioBurst_T* const Bursts, const uint16_t numTxBursts, const uint32_t frequency) {
164  Cpu.spiInit();
165 
166  Cpu.initTimer();
167  setTxPwrReg(txPower);
168 
169  // Give the system the time of four bits to initialize everything (approx. 10ms)
170  Cpu.addTimerDelay(4);
171  Cpu.startTimer();
172 
173  for (uint16_t burstIdx = 0; burstIdx < numTxBursts; ++burstIdx) {
174  Cpu.resetWatchdog();
175 
176  // Special handling in case of zero length bursts
177  if (Bursts[burstIdx].getBurstLength() == 0) {
178  Cpu.waitTimer();
179  if (burstIdx + 1 < numTxBursts) {
180  Cpu.addTimerDelay((int16_t)Bursts[burstIdx].get_T_RB() - Bursts[burstIdx].getBurstLength());
181  }
182  continue;
183  }
184 
185  // Calculate frequency register setting
186  const uint32_t cFrequency = (uint32_t) Bursts[burstIdx].getCarrierOffset();
187  const uint32_t modFreq = frequency + cFrequency;
188  Cpu.waitTimer();
189  setFrequencyReg(modFreq);
190 
191  const uint8_t* burstData = Bursts[burstIdx].getBurst();
192  for (uint8_t byteIdx = 0; byteIdx < Bursts[burstIdx].getBurstLengthBytes(); ++byteIdx) {
193  //TODO Write data bytes to FIFO
194  }
195 
196  Cpu.addTimerDelay(2);
197  Cpu.waitTimer();
198 
199  // Start TX, you shall ensure that the TX stops the TX after the data has been transmitted
200  // Otherwise the system will transmit for a long time if it crashes somewhere
201  setMode(RFM69_MODE_TX);
202 
203  Cpu.addTimerDelay(Bursts[burstIdx].getBurstLength());
204  Cpu.waitTimer();
205  setMode(RFM69_MODE_SLEEP);
206 
207  // Wait for the end of the burst
208  if (burstIdx <= numTxBursts) {
209  Cpu.addTimerDelay((int16_t)Bursts[burstIdx].get_T_RB() - 2);
210  }
211  }
212 
213 
214  setMode(RFM69_MODE_SLEEP);
215  Cpu.stopTimer();
216  Cpu.spiDeinit();
217 
218  return 0;
219  }
