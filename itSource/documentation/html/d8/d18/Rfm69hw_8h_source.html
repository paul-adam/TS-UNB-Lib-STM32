211  int16_t init(void) {
212  // TODO check if chips is actually present and return negative value in case of error
213  Cpu.spiInit();
214 
215  // Read register 0x0c and check if 0x02, if no there is no chip
216  {
217  uint8_t spiData[2]={0x0c, 0};
218  Cpu.spiSendReceive(spiData, 2);
219  if (spiData[1] != 0x02) {
220  // Chip not found!
221  Cpu.spiDeinit();
222 
223  return -1;
224  }
225  }
226 
227  /*
228  * Initialize with presets
229  */
230  const uint8_t presets[] = RFM69_CONFIG_ARRAY;
231  uint8_t idx = 0;
232  while (presets[idx] != 0) {
233  Cpu.spiSend(&presets[idx + 1], presets[idx]);
234  idx += presets[idx] + 1;
235  }
236  setMode(RFM69_MODE_SLEEP);
237 
238  Cpu.spiDeinit();
239 
240  return 0;
241  }

244  int16_t transmit(const RadioBurst_T* const Bursts, const uint16_t numTxBursts, const uint32_t frequency) {
245  Cpu.spiInit();
246 
247  Cpu.initTimer();
248  setTxPwrReg(txPower);
249 
250  // Give the system the time of four bits to initialize everything (approx. 10ms)
251  Cpu.addTimerDelay(4);
252  Cpu.startTimer();
253 
254  for (uint16_t burstIdx = 0; burstIdx < numTxBursts; ++burstIdx) {
255  Cpu.resetWatchdog();
256 
257  // Special handling in case of zero length bursts
258  if (Bursts[burstIdx].getBurstLength() == 0) {
259  Cpu.waitTimer();
260  if (burstIdx + 1 < numTxBursts) {
261  Cpu.addTimerDelay((int16_t)Bursts[burstIdx].get_T_RB() - Bursts[burstIdx].getBurstLength());
262  }
263  continue;
264  }
265 
266  const uint32_t cFrequency = (uint32_t) Bursts[burstIdx].getCarrierOffset();
267  const uint32_t modFreq = frequency + cFrequency;
268  Cpu.waitTimer();
269  setFrequencyReg(modFreq);
270 
271  const uint8_t* burstData = Bursts[burstIdx].getBurst();
272  for (uint8_t byteIdx = 0; byteIdx < Bursts[burstIdx].getBurstLengthBytes(); ++byteIdx) {
273  uint8_t data[2] = {0x80, burstData[byteIdx]};
274  Cpu.spiSend(data, 2);
275  }
276 
277  // Add one dummy byte, if this byte is actually transmitted, the TX switches itself into sleep mode
278  // because we did not set the sleep command
279  {
280  uint8_t data[2] = {0x80, 0};
281  Cpu.spiSend(data, 2);
282  }
283  setMode(RFM69_MODE_FS);
284 
285  Cpu.addTimerDelay(2);
286  Cpu.waitTimer();
287  setMode(RFM69_MODE_TX);
288 
289  Cpu.addTimerDelay(Bursts[burstIdx].getBurstLength());
290  Cpu.waitTimer();
291  setMode(RFM69_MODE_SLEEP);
292 
293  /*
294  * If we are not in the last burst wait for the next burst to start.
295  * If we are in the last burst we do not have to restart the counter again.
296  *
297  * We wake up 2 bits before the new burst starts. This gives us enough time to shift the
298  * data into the FIFO before the next transmission starts.
299  */
300  if (burstIdx <= numTxBursts) {
301  Cpu.addTimerDelay((int16_t)Bursts[burstIdx].get_T_RB() - Bursts[burstIdx].getBurstLength() - 2);
302  }
303  }
304  //Cpu.waitTimer();
305  setMode(RFM69_MODE_SLEEP);
306  Cpu.stopTimer();
307  Cpu.spiDeinit();
308 
309  return 0;
310  }

333  void setFrequencyReg(const uint32_t frequency) {
334  /*
335  * Union to simplify byte access.
336  */
337  union splituint32 {
338  uint8_t bytes[4];
339  uint32_t single;
340  };
341  union splituint32 regVal;
342  regVal.single = frequency;
343  uint8_t data[4] = {
344  RFM69_WRITE_FRF, regVal.bytes[2],
345  regVal.bytes[1], regVal.bytes[0]};
346  Cpu.spiSend(data, 4);
347  }

362  int8_t setTxPwrReg(int8_t power) {
363  // Special handling if boost pin is used
364  if (BOOST_PIN) {
365  // Limit to valid power range
366  if (power > 17)
367  power = 17;
368  if (power < -2)
369  power = -2;
370 
371  // If less than 13dBm use PA1 only, otherwise use PA1 and PA2
372  if (power <= 13) {
373  uint8_t regPower = power + 18;
374  uint8_t data[2] = {RFM69_WRITE_PA_LEVEL,(uint8_t) (RFM69_PA1_ON | regPower)};
375  Cpu.spiSend(data, 2);
376  }
377  else {
378  // Use PA1 and PA2
379  uint8_t regPower = power + 14;
380 
381  uint8_t data[2] = {RFM69_WRITE_PA_LEVEL, (uint8_t) (RFM69_PA1_ON | RFM69_PA2_ON | regPower)};
382  Cpu.spiSend(data, 2);
383  }
384  return power;
385  }
386  else {
387  // PA Boost Pin is not used
388 
389  // Limit to valid power range
390  if (power > 13)
391  power = 13;
392  if (power < -18)
393  power = -18;
394 
395  uint8_t regPower = power + 18;
396 
397  uint8_t data[2] = {RFM69_WRITE_PA_LEVEL, (uint8_t) (RFM69_PA0_ON + regPower)};
398  Cpu.spiSend(data, 2);
399 
400  return power;
401  }
402  }
